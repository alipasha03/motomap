\section{Graf ve Feature Modeli}
Yol a\u{g}\i{} yonlu agirlikli graf:
\begin{equation}
G=(V,E), \quad E \subseteq V\times V.
\end{equation}

Kenar ozellikleri:
\begin{equation}
\mathbf{x}_e = [L_e, v^{max}_e, grade_e, curve_e, risk_e, toll_e, road\_type_e].
\end{equation}

\subsection{Temel sure}
\begin{equation}
T_e = \frac{L_e}{\max\left(1,\frac{v^{max}_e\cdot\phi}{3.6}\right)} + \delta,
\end{equation}
\(\phi=\texttt{MOTOMAP\_SPEED\_FACTOR},\;\delta=\texttt{MOTOMAP\_SEGMENT\_DELAY\_S}\).

\subsection{Viraj ve risk}
\begin{align}
N_{fun}(e)&=\#\{\theta:15^\circ<\theta<45^\circ\},\\
N_{danger}(e)&=\#\{\theta:\theta>60^\circ\},\\
S_{curve}(e)&=\frac{N_{fun}(e)}{1+N_{danger}(e)},\\
risk_e&=\mathbb{I}[hairpin(e)>0\land grade_e<-0.08].
\end{align}

\section{Mod Bazli Maliyet Fonksiyonlari}
\noindent Kodda her kenar once gecis suresine, sonra mod-bazli ceza/odul katsayilarina donusturulur.

\subsection{Standart}
\begin{equation}
W^{std}_e = T_e\,(1+p_{road}(e)).
\end{equation}

\subsection{Viraj keyfi}
\begin{align}
B_e &= 1 + 0.55S_{curve}(e) + 0.14N_{fun}(e) + b_{road}(e),\\
P_e &= 1 + 0.12N_{danger}(e) + 0.45risk_e,\\
W^{fun}_e &= \frac{T_e}{\max(10^{-6},B_e)}\,P_e.
\end{align}

\subsection{Guvenli}
\begin{align}
d_e &= \max(0,|\min(0,grade_e)|-0.08),\\
P^{safe}_e &= 1 + 0.55N_{danger}(e)+1.40risk_e+5.00d_e+p^{safe}_{road}(e),\\
W^{safe}_e &= T_e\,P^{safe}_e.
\end{align}

\section{Kod Karsiligi: Tek Satirlik Matematiksel Ozet}
\noindent Kodun cekirdegi su optimizasyon problemidir:
\begin{equation}
\pi_m^\star=\arg\min_{\pi \in \mathcal{P}(s,t)} \sum_{e \in \pi} W_e^{(m)},
\qquad
m \in \{\text{standart},\text{viraj},\text{guvenli}\}.
\end{equation}

\noindent Burada
\begin{equation}
W_e^{(m)} = T_e \cdot \alpha_e^{(m)} \cdot \beta_e^{(m)},
\end{equation}
\noindent $\alpha_e^{(m)}$ yol tipi ve egim etkisini, $\beta_e^{(m)}$ ise viraj, risk ve ucretli yol tercihlerini temsil eder.

\section{Website Uzerinden Google--Valhalla--Diger Motor Karsilastirmasi}
\noindent Website ayni baslangic-varis cifti icin Google veya Valhalla baz rotasini ve MOTOMAP modlarini ayni harita uzerinde cizer; karsilastirma metrikleri:
\begin{align}
R_d &= \frac{D_{\text{engine}}}{D_{\text{base}}}, &
R_t &= \frac{T_{\text{engine}}}{T_{\text{base}}},\\
\Delta_{fun} &= F_{\text{viraj}} - F_{\text{standart}}, &
\Delta_{risk} &= K_{\text{guvenli}} - K_{\text{standart}}.
\end{align}

\noindent Website PASS kosulu cok kisadir:
\begin{equation}
\Delta_{fun}\ge 0,\qquad \Delta_{risk}\le 0,\qquad
0.7 \le R_d \le 1.4,\qquad 0.5 \le R_t \le 1.8.
\end{equation}

\noindent Google genel-surucu optimumunu, Valhalla OSM-tabanli genel amacli maliyeti, MOTOMAP ise motosiklet-ozel agirlikli optimizasyonu temsil eder; bu nedenle ayni OD cifti icin rota geometrisi ve risk profili farkli cikar.

\section{Algoritmalar (Adim Adim)}

\subsection{Algoritma-1: Feature graph olusturma}
\begin{algorithm}[h]
\caption{BuildFeatureGraph}
\begin{algorithmic}[1]
\Require Raw graph $G$
\State $G \leftarrow$ legal olmayan kenarlari filtrele
\State node elevation ekle, edge grade hesapla
\State eksik lanes/maxspeed/surface alanlarini doldur
\State viraj acilarindan $N_{fun},N_{danger},S_{curve}$ cikar
\State her edge icin $risk_e$ isaretle
\State \Return $G$
\end{algorithmic}
\end{algorithm}

\subsection{Algoritma-2: Mod bazli kenar agirligi}
\begin{algorithm}[h]
\caption{ComputeWeightByMode}
\begin{algorithmic}[1]
\Require edge $e$, mode $m \in \{std,fun,safe\}$
\State taban sure $T_e$ hesapla
\If{$m=std$}
  \State \Return $W^{std}_e$
\ElsIf{$m=fun$}
  \State \Return $W^{fun}_e$
\Else
  \State \Return $W^{safe}_e$
\EndIf
\end{algorithmic}
\end{algorithm}

\subsection{Algoritma-3: Ucretli/Ucretsiz secim}
\begin{algorithm}[h]
\caption{SelectRouteWithTollPreference}
\begin{algorithmic}[1]
\Require source $s$, target $t$, preference $p$
\State $\pi_{serbest} \leftarrow \text{Dijkstra}(G,s,t)$
\State $\pi_{ucretsiz} \leftarrow \text{Dijkstra}(G\setminus E_{toll},s,t)$
\If{$p=\texttt{ucretsiz}$}
  \State \Return $\pi_{ucretsiz}$
\Else
  \State \Return $\pi_{serbest}$
\EndIf
\end{algorithmic}
\end{algorithm}

\subsection{Algoritma-4: Dijkstra cekirdegi}
\begin{algorithm}[h]
\caption{Dijkstra}
\begin{algorithmic}[1]
\Require $G=(V,E)$, $s$, $t$, positive edge weights
\State $dist[v]\gets\infty$, $dist[s]\gets 0$
\State priority queue $Q\gets\{(0,s)\}$
\While{$Q$ not empty}
  \State extract $(d,u)$ with min key
  \If{$u=t$} \State \textbf{break} \EndIf
  \For{each edge $(u,v)$}
    \State $alt\gets d+W_{uv}$
    \If{$alt<dist[v]$}
      \State update $dist[v], parent[v]$
      \State push $(alt,v)$
    \EndIf
  \EndFor
\EndWhile
\State reconstruct path using $parent$
\end{algorithmic}
\end{algorithm}

\section{Dogruluk ve Karmasiklik}
\noindent Tum kenar agirliklari pozitif oldugunda Dijkstra optimal sonucu verir. Binary heap ile karmasiklik
\begin{equation}
\mathcal{O}((|V|+|E|)\log|V|)
\end{equation}
\noindent olur. Bu nedenle maliyet fonksiyonunda negatif agirlik uretmemek, algoritmanin hem matematiksel dogrulugu hem de kod kararliligi icin zorunludur.

\section{Akilli Motosiklet Navigasyonu: Daha Yesil ve Surdurulebilir Sehirler}
\noindent Sehir ici ulasim yalnizca hiz problemi degil, enerji ve atik problemidir. Motosikletin binek araca gore cok daha dusuk kutle ile hareket etmesi, birim yolcu basina tasinan olu kutleyi azaltir; bu nedenle ayni ulasim isi daha dusuk enerji ile tamamlanir. Hafif arac secimi, yol asinmasi, alan isgali ve toplam karbon ayak izi uzerinde dogrudan olumlu etki yaratir.

\begin{center}
\begin{tabular}{@{}lll@{}}
\toprule
Arac tipi & Ortalama agirlik & Cevresel sonuc \\
\midrule
Binek arac & 1500--2000 kg & Yuksek enerji tuketimi ve yuksek alan isgali \\
Motosiklet & 150--200 kg & Dusuk enerji tuketimi ve dusuk karbon ayak izi \\
\bottomrule
\end{tabular}
\end{center}

\noindent Bununla birlikte surdurulebilirlik sadece arac secmekle bitmez; rota secmek de gerekir. Lane filtering modele dahil edildiginde motosiklet gereksiz dur-kalk yerine akisa yakin bir hareket profili izler, rolanti suresi kisalir ve egzoz emisyonu azalir. Dogru ETA, sehir genelinde daha verimli rota dagilimi yaratir. Surucuden surucuye yardimlasma mantigi ise tekil arizalarda agir cekici ihtiyacini azaltarak dolayli karbon maliyetini de dusurur.

\noindent Topografik optimizasyon ve CC uyumu mekanik atigi azaltan ikinci eksendir. Dusuk hacimli motorun dik egimli segmente zorlanmadan daha dengeli profilde yonlendirilmesi, motor yagi bozulmasini ve termal yuklenmeyi azaltir. Yol kalitesi ve viraj analizi, ani frenleme ve sert ivmelenmeyi dusurdugu icin balata tozu ve lastik mikro-plastigi uretimini de sinirlar. Bu nedenle akilli motosiklet navigasyonu yalnizca daha hizli degil, daha temiz bir ulasim aracidir.

\section{Motosiklet Navigasyonunun Gizli Dunyasi: Lane Filtering ve Curviness Index}
\noindent Standart haritalar trafik akislarini otomobil merkezli maliyet fonksiyonlariyla yorumlar; bu nedenle motosiklet davranisini eksik temsil eder. Motosiklet icin trafik, tam tikanma degil, manevra katsayisinin devreye girdigi farkli bir dinamik rejimdir. MOTOMAP bu farki, baz sure $T_e$ uzerine eklenen profil-katsayilari ile sayisallastirir ve ETA sapmasini azaltir.

\noindent Curviness Index ise keyifli rotayi geometri, egim ve riskin ortak bileşkesi olarak tanimlar. Bu yaklasimda yalnizca kac viraj oldugu degil, virajin acisi, yogunlugu ve iniş riski de hesaba katilir. Viraj keyfi modunda bonus terimi maliyeti asagi cekerken, tehlike ve yuksek risk maliyeti tekrar yukari iter; guvenli mod ise ayni geometriyi ters okuyup riskli segmente daha buyuk ceza verir.

\noindent Bu nedenle lane filtering zamani, curviness ise surus karakterini optimize eder. Google ve Valhalla genel amacli route engine olarak guclu tabanlar sunsa da, motosiklet-ozel karar katmani eklenmediginde surucunun gercek davranisi eksik temsil edilir. MOTOMAP'in farki, bu iki katmani ayni maliyet denkleminde birlestirerek zamani, keyfi, guvenligi ve surdurulebilirligi birlikte optimize etmesidir.
